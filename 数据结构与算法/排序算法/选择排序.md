### 特点
时间复杂度 `O(n²)`  
最好情况 `O(n²)`    
最差情况  `O(n²)`   
空间复杂度 `O(1)`     
不稳定排序 

```js
let arr = [1, 4, 5, 6, 5, 6, 3, 6, 3];
let sort = (arr) => {
    for (let i = 0; i < arr.length - 1; i++) {
        let minIndex = i;
        for (let j = i + 1; j < arr.length; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }
        swap(arr, i, minIndex);
    }
    return arr;
};
let swap = (arr, a, b) => {
    if (a === b) {
        return;
    }
    arr[a] = arr[a] ^ arr[b];
    arr[b] = arr[a] ^ arr[b];
    arr[a] = arr[a] ^ arr[b];
};

console.log(sort(arr))       // [1, 3, 3, 4, 5, 5, 6, 6, 6] 

```
