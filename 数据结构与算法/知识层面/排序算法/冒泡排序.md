### 特点
时间复杂度 `O(n²)`  
最好情况 `O(n)`    
最差情况  `O(n²)`   
空间复杂度 `O(1)`     
稳定排序 

```js
var arr = Array.from({ length: 20 }, () => Math.round(Math.random() * 50));

function bubble(arr) {
    for (let i = 0; i < arr.length - 1; i++) {
        for (let j = 0; j < arr.length - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                [arr[j],arr[j+1]] = [arr[j+1],arr[j]]
            }
        }
    }
    return arr;
}

bubble(arr);
console.log(bubble(arr)); 

```
